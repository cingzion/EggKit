3、配置模板引擎



	1、npm i egg-view-ejs --save

	


	2.找到 {app_root}/config/plugin.js


	exports.ejs = {
 	 enable: true,
 	 package: 'egg-view-ejs',
	};


	3.找到 {app_root}/config/config.default.js


	//配置ejs模板引擎
  	config.view = {
   	 mapping: {
    	  '.html': 'ejs',
   	 }
  	};
